```js
  function _readEnv() {
    let raw = null;
    try { raw = storage.getItem(KEY); } catch (_) { return { v: VERSION, profiles: [] }; }
    if (!raw) return { v: VERSION, profiles: [] };
    let env;
    try { env = JSON.parse(raw); } catch (_) { return { v: VERSION, profiles: [] }; }
    if (!env || typeof env !== 'object' || env.v !== VERSION) return { v: VERSION, profiles: [] };
    if (!Array.isArray(env.profiles)) env.profiles = [];
    return env;
  }

  function _writeEnv(env) {
    try { storage.setItem(KEY, JSON.stringify(env)); return { ok: true }; }
    catch (e) { return { ok: false, error: (e && e.name === 'QuotaExceededError') ? 'quota' : 'storage' }; }
  }

  function _tuningOf(env) {
    const t = (env.tuning && typeof env.tuning === 'object') ? env.tuning : {};
    return {
      accepted: Array.isArray(t.accepted) ? t.accepted : [],
      dismissed: Array.isArray(t.dismissed) ? t.dismissed : [],
    };
  }

  function getTuning() { return _tuningOf(_readEnv()); }

  function _clampSum(ops, clampMin, clampMax) {
    const sum = ops.reduce((a, o) => a + (Number(o.step) || 0), 0);
    return Math.min(clampMax, Math.max(clampMin, Math.round(sum * 100) / 100));
  }

  function addTuningOp(pairKey, offsetKey, unit, op) {
    const env = _readEnv();
    const tuning = _tuningOf(env);
    let entry = tuning.accepted.find(e => e.pairKey === pairKey && e.offsetKey === offsetKey);
    if (!entry) {
      entry = { pairKey, offsetKey, unit, value: 0,
                clampMin: op.clampMin ?? -Infinity, clampMax: op.clampMax ?? Infinity, ops: [] };
      // -Infinity/Infinity never actually stored: accept ops always carry clamps (harvest only
      // offers clamped suggestions); revert-first cannot happen (nothing to revert).
      tuning.accepted.push(entry);
    }
    if (op.clampMin != null) entry.clampMin = op.clampMin;
    if (op.clampMax != null) entry.clampMax = op.clampMax;
    // Clamp defense (S5): reject accepts whose raw sum is already at/past the clamp
    // in the step's direction, so the raw sum can never outrun a revert.
    if (op.kind !== 'revert') {
      const rawSum = entry.ops.reduce((a, o) => a + (Number(o.step) || 0), 0);
      const next = rawSum + (Number(op.step) || 0);
      if (next < entry.clampMin || next > entry.clampMax) return { ok: false, error: 'clamp' };
    }
    entry.ops.push({
      opId: _newId(), kind: op.kind === 'revert' ? 'revert' : 'accept',
      step: Number(op.step) || 0,
      ...(op.symptomId ? { symptomId: op.symptomId } : {}),
      date: _now(),
    });
    entry.value = _clampSum(entry.ops, entry.clampMin, entry.clampMax);
    env.tuning = tuning;
    const w = _writeEnv(env);
    return w.ok ? { ok: true, entry } : w;
  }

  function dismissSuggestion(key) {
    const env = _readEnv();
    const tuning = _tuningOf(env);
    const existing = tuning.dismissed.find(d => d.key === key);
    if (existing) existing.date = _now();
    else tuning.dismissed.push({ key, date: _now() });
    env.tuning = tuning;
    return _writeEnv(env);
  }

  function getDismissal(key) {
    return getTuning().dismissed.find(d => d.key === key) || null;
  }

  function revertTuning(pairKey, offsetKey) {
    const env = _readEnv();
    const tuning = _tuningOf(env);
    const entry = tuning.accepted.find(e => e.pairKey === pairKey && e.offsetKey === offsetKey);
    if (!entry) return { ok: false, error: 'not-found' };
    const rawSum = entry.ops.reduce((a, o) => a + (Number(o.step) || 0), 0);
    entry.ops.push({ opId: _newId(), kind: 'revert', step: -rawSum, date: _now() });
    entry.value = _clampSum(entry.ops, entry.clampMin, entry.clampMax);
    env.tuning = tuning;
    const w = _writeEnv(env);
    return w.ok ? { ok: true, entry } : w;
  }
```

```js
    try { env = JSON.parse(json); } catch (_) { return { ok: false, error: 'parse' }; }
    if (!env || typeof env !== 'object' || env.v !== VERSION || !Array.isArray(env.profiles)) {
      return { ok: false, error: 'format' };
    }
    const incoming = env.profiles.filter(p =>
      p && typeof p === 'object' && p.id && p.state && typeof p.state === 'object');
    const byId = new Map();
    _read().forEach(p => byId.set(p.id, p));
    incoming.forEach(p => byId.set(p.id, p));

    // Merge tuning (op-union) into a copy of current tuning — no writes yet.
    const cur = _tuningOf(_readEnv());
    const incTuning = (env.tuning && typeof env.tuning === 'object') ? env.tuning : {};
    (Array.isArray(incTuning.accepted) ? incTuning.accepted : []).forEach(inc => {
      if (!inc || !inc.pairKey || !inc.offsetKey || !Array.isArray(inc.ops)) return;
      let entry = cur.accepted.find(e => e.pairKey === inc.pairKey && e.offsetKey === inc.offsetKey);
      if (!entry) { entry = { ...inc, ops: [] }; cur.accepted.push(entry); }
      const seen = new Set(entry.ops.map(o => o.opId));
      inc.ops.forEach(o => { if (o && o.opId && !seen.has(o.opId)) { entry.ops.push(o); seen.add(o.opId); } });
      entry.ops.sort((a, b) => String(a.date).localeCompare(String(b.date)));
      entry.value = _clampSum(entry.ops, entry.clampMin, entry.clampMax);
    });
    (Array.isArray(incTuning.dismissed) ? incTuning.dismissed : []).forEach(inc => {
      if (!inc || !inc.key) return;
      const ex = cur.dismissed.find(d => d.key === inc.key);
      if (!ex) cur.dismissed.push({ key: inc.key, date: inc.date || _now() });
      else if (String(inc.date) > String(ex.date)) ex.date = inc.date;
    });

    const out = _readEnv();
    out.profiles = [...byId.values()];
    if (cur.accepted.length || cur.dismissed.length) out.tuning = cur;
    const w = _writeEnv(out);                       // SINGLE write — atomic
    return w.ok ? { ok: true, count: incoming.length } : w;
  }
```

**Harvest algorithm (implement exactly — spec §3.1–3.2; all dates below are from the FIRING nOZZLE BUCKET, plan-review M3):**
1. Group all profiles by `pairKey = state.printer + '|' + state.material`; skip pairs where `printerExists`/`materialExists` is false (journal untouched, no suggestion).
2. Per pair, per symptom: bucket failed outcomes tagged with the symptom BY the profile's `state.nozzle`. The **firing bucket** is the nozzle bucket with ≥2 failures (if several qualify, the one with the newest failure). `bucketLastFail` = newest failure date IN the firing bucket — every subsequent date comparison uses `bucketLastFail`, never the symptom-wide newest (a stray newer failure on another nozzle must not refresh gates for this bucket). `workedNewest` = newest worked-outcome date for the pair.
3. Fire test: a firing bucket exists AND NOT (`workedNewest > bucketLastFail`) (positive lock-out — deliberately pair-wide, spec §3.2).
4. For each fired symptom: `rulesForSymptom(symptomId, materialGroup(materialId))` → if primary is null, nothing; if primary `kind:'advice'` → advice card; else offset candidate. Attach `secondaryHints = secondaries.map(r => r.why)` for the card copy.
5. Suppressions (apply to offset AND advice candidates — advice/conflict cards are dismissible too, plan-review S1): (a) dismissal exists for the candidate's key AND `bucketLastFail <= dismissal.date`; (b) **anti-ride, per-symptom scope (plan-review M2):** an accept op for pair+offsetKey **with the SAME `symptomId`** exists with `op.date >= bucketLastFail`; (c) cumulative value already at the rule's clamp in the step's direction (clamp stop).
6. Contradiction pass over surviving offset candidates: group by pair+offsetKey; if both directions present, replace ALL candidates for that key with one `kind:'conflict'` card carrying `conflictingSymptoms` and key `printerId|materialId|conflict|offsetKey` (dismissible under rule 5a with `bucketLastFail` = newest across its contributing buckets).
7. Emit: `key = printerId|materialId|symptomId|offsetKey` (advice: `printerId|materialId|symptomId|adviceKey`), `step`, `unit`, `clampMin`, `clampMax` (from the rule — `accept()` forwards them to `addTuningOp`, plan-review S2), `cumulativeAfterAccept = clamp(current + step)`, `secondaryHints`, `evidence = {failed, worked, lastDate: bucketLastFail, nozzles: [firing bucket nozzle], environments}`.
